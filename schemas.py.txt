from typing import List, Optional
from pydantic import BaseModel, Field, model_validator
from enum import Enum

class ActionCategory(str, Enum):
    MIGRATION = "migration"
    DEBUG = "debug"
    ENHANCE = "enhance"
    REFACTOR = "refactor"
    TESTING = "testing"
    DEPLOYMENT = "deployment"
    RESEARCH = "research"
    OTHER = "other"
    DOCUMENTATION = "documentation"

class DetailLevel(str, Enum):
    WITH_DETAILS = "with_details"
    MIDDLE = "middle"
    NO_DETAILS = "no_details"

class Who(BaseModel):
    identified: bool = Field(..., description="True if a specific role or team is mentioned.")
    evidence: Optional[str] = Field(None, description="Verbatim quote of the requester.")

class What(BaseModel):
    identified: bool = Field(..., description="True if a technical action is stated.")
    category: ActionCategory = Field(..., description="The primary technical category.")
    intent_evidence: Optional[str] = Field(None, description="Verbatim quote of the technical intent.")
    the_level_of_details_in_intent: DetailLevel = Field(..., description="Technical detail granularity.")

class Why(BaseModel):
    identified: bool = Field(..., description="True if business value is explicitly stated.")
    value_evidence: Optional[str] = Field(None, description="Verbatim quote of the justification.")

class CustomerImpact(BaseModel):
    identified: bool = Field(..., description="True if end-user impact is mentioned.")
    impact_evidence: Optional[str] = Field(None, description="Verbatim quote of the impact.")

class AgilePrincipleCheck(BaseModel):
    presence_ac: bool = Field(..., description="True if testable acceptance criteria exist.")
    ac_evidence: Optional[str] = Field(None, description="Verbatim quote of the extracted AC.")
    
    # 3-part feedback constraint as requested
    english_text_feedback: str = Field(
        ..., 
        description=(
            "Strict 3-part feedback in max 10 words total. Format: [Grammar] | [Length] | [Vocab/Typos]. "
            "Example: 'Good grammar | Sentences long | Plain vocab, no typos.'"
        )
    )
    overal_feedback: str = Field(..., description="Give a overal feedback on the jira desription, a good jira desription should align wit Agile Standard, short, plain, having Who, What, Why (value) and Acceptence Criterira")

class JiraAnalysis(BaseModel):
    reasoning: str = Field(..., description="Internal logic for the classification.")
    who: Who
    what: What
    why: Why
    customer_impact: CustomerImpact
    ac_defined: AgilePrincipleCheck
    
    grooming_questions: List[str] = Field(default_factory=list, description="4 technical blocker questions.")
    agile_standard: bool = Field(default=False, description="Calculated locally based on 3Ws and AC.")
    org_text: str = Field(default="", description="Original input text populated locally.")

    @model_validator(mode='after')
    def enforce_data_integrity(self) -> 'JiraAnalysis':
        """
        Lead AI Engineer Guard:
        This validator acts as a final filter. If 'What' intent is not identified,
        it wipes any hallucinated questions from the LLM response to ensure 100% data accuracy.
        """
        if not self.what.identified:
            self.grooming_questions = []
        
        return self
