from typing import List, Optional
from pydantic import BaseModel, Field
from enum import Enum

class ActionCategory(str, Enum):
    MIGRATION = "migration"
    DEBUG = "debug"
    ENHANCE = "enhance"
    REFACTOR = "refactor"
    TESTING = "testing"
    DEPLOYMENT = "deployment"
    RESEARCH = "research"
    OTHER = "other"
    DOCUMENTATION = "documentation"

class DetailLevel(str, Enum):
    WITH_DETAILS = "with_details"
    MIDDLE = "middle"
    NO_DETAILS = "no_details"

class Who(BaseModel):
    identified: bool = Field(..., description="True if role/team is explicitly mentioned.")
    evidence: Optional[str] = Field(None, description="Verbatim quote of the requester.")

class What(BaseModel):
    identified: bool = Field(..., description="True if technical action is stated.")
    category: ActionCategory = Field(..., description="Primary technical category.")
    intent_evidence: Optional[str] = Field(None, description="Verbatim quote of the intent.")
    the_level_of_details_in_intent: DetailLevel = Field(..., description="Technical detail level.")

class Why(BaseModel):
    identified: bool = Field(..., description="True if business value is stated.")
    value_evidence: Optional[str] = Field(None, description="Verbatim quote of justification.")

class CustomerImpact(BaseModel):
    identified: bool = Field(..., description="True if end-user impact is mentioned.")
    impact_evidence: Optional[str] = Field(None, description="Verbatim quote of impact.")

class AgilePrincipleCheck(BaseModel):
    presence_ac: bool = Field(..., description="True if testable AC exists.")
    ac_evidence: Optional[str] = Field(None, description="Verbatim AC extraction.")
    
    # Updated field with strict 3-part constraint
    english_text_feedback: str = Field(
        ..., 
        description=(
            "Strict 3-part feedback in max 10 words. Format: [Grammar] | [Length] | [Vocab/Typos]. "
            "Example: 'Good grammar | Sentences long | Simple vocab, no typos.'"
        )
    )
    overal_feedback: str = Field(..., description="Feedback on structural completeness.")

class JiraAnalysis(BaseModel):
    reasoning: str = Field(..., description="Internal extraction logic.")
    who: Who
    what: What
    why: Why
    customer_impact: CustomerImpact
    ac_defined: AgilePrincipleCheck
    grooming_questions: List[str] = Field(..., description="4 technical blocker questions.")
    
    # Local-only fields
    agile_standard: bool = Field(default=False)
    org_text: str = Field(default="")
